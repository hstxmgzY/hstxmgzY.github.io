    找出数组中的第 K 大整数

    Problem: 1985. 找出数组中的第 K 大整数

    思路

    就是很简单的排序，因为本来是在nk上刷的，要求用快排 然后就用快排实现了一下

    解题方法

    快速排序

    复杂度

    时间复杂度:

    \(O(nlogn)\)

    空间复杂度:

    \(O(1)\)

    Code

    慢吞吞的快排（击败个位数百分比）

    class Solution:
    def kthLargestNumber(self, nums: List[str], k: int) -> str:
        def qs(num):
            n = len(num)
            if n <= 1:
                return num
            q = num[n // 2]
            l = [x for x in num if int(x) < int(q)]
            m = [x for x in num if int(x) == int(q)]
            r = [x for x in num if int(x) > int(q)]
            return qs(l) + m + qs(r)
        asort = qs(nums)
        print(asort)
        n = len(nums)
        return asort[n - k]

    class Solution:
    def kthLargestNumber(self, nums: List[str], k: int) -> str:

        def qs(num,k):
            n = len(num)
            if n <= 1 :
                return num[0]
            q = num[n // 2]
            l = [x for x in num if int(x) < int(q)]
            m = [x for x in num if int(x) == int(q)]
            r = [x for x in num if int(x) > int(q)]
            if len(l) < k <= len(l) + len(m):
                return q
            elif k <= len(l):
                return qs(l,k)
            else:
                k = k - len(l)-len(m)
                return qs(r,k)
        k = len(nums) - k + 1
        return qs(nums,k)

    堆排序

    class Solution:
    def kthLargestNumber(self, nums: List[str], k: int) -> str:
        heap = []
        n = len(nums)
        for c in nums:
            heapq.heappush(heap, int(c))
        for _ in range(n-k):
            heapq.heappop(heap)
        ans = heapq.heappop(heap)
        return str(ans)
